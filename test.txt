-- 2. إعدادات النظام
local Settings = {
    Enabled = false,
    FOV = 60,
    TeamCheck = true,
    VisibilityCheck = true,
    TargetPart = "Head",
    Prediction = true,
    Smoothing = 0.5
}

-- 3. عناصر التحكم في GUI
local function CreateToggle(name, yPos, configKey)
    local ToggleFrame = Instance.new("Frame")
    ToggleFrame.Size = UDim2.new(1, -20, 0, 30)
    ToggleFrame.Position = UDim2.new(0, 10, 0, yPos)
    ToggleFrame.BackgroundTransparency = 1
    ToggleFrame.Parent = MainFrame

    local ToggleButton = Instance.new("TextButton")
    ToggleButton.Size = UDim2.new(0, 20, 0, 20)
    ToggleButton.Position = UDim2.new(1, -20, 0, 5)
    ToggleButton.Text = ""
    ToggleButton.BackgroundColor3 = Settings[configKey] and Color3.new(0, 1, 0) or Color3.new(1, 0, 0)
    ToggleButton.Parent = ToggleFrame

    local ToggleLabel = Instance.new("TextLabel")
    ToggleLabel.Text = name
    ToggleLabel.Size = UDim2.new(1, -30, 1, 0)
    ToggleLabel.TextColor3 = Color3.new(1, 1, 1)
    ToggleLabel.BackgroundTransparency = 1
    ToggleLabel.TextXAlignment = Enum.TextXAlignment.Left
    ToggleLabel.Parent = ToggleFrame

    ToggleButton.MouseButton1Click:Connect(function()
        Settings[configKey] = not Settings[configKey]
        ToggleButton.BackgroundColor3 = Settings[configKey] and Color3.new(0, 1, 0) or Color3.new(1, 0, 0)
    end)

    return ToggleButton
end

-- إنشاء عناصر التحكم

-- 4. نظام Silent AIM الأساسي
local CurrentTarget = nil
local FOVCircle = Drawing.new("Circle")
FOVCircle.Visible = false
FOVCircle.Color = Color3.new(1, 0, 0)
FOVCircle.Thickness = 1
FOVCircle.NumSides = 64
FOVCircle.Filled = false

local function FindTarget()
    if not Settings.Enabled then return nil end
    
    local closestTarget = nil
    local closestAngle = math.rad(Settings.FOV)
    local cameraPos = Camera.CFrame.Position
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player == LocalPlayer then continue end
        if Settings.TeamCheck and player.Team == LocalPlayer.Team then continue end
        
        local character = player.Character
        if not character then continue end
        
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 then continue end
        
        local targetPart = character:FindFirstChild(Settings.TargetPart)
        if not targetPart then continue end
        
        -- حساب الزاوية
        local direction = (targetPart.Position - cameraPos).Unit
        local angle = math.acos(Camera.CFrame.LookVector:Dot(direction))
        
        -- التحقق من الرؤية
        if Settings.VisibilityCheck then
            local raycastParams = RaycastParams.new()
            raycastParams.FilterDescendantsInstances = {LocalPlayer.Character, character}
            raycastParams.FilterType = Enum.RaycastFilterType.Blacklist
            local raycastResult = workspace:Raycast(cameraPos, direction * 1000, raycastParams)
            
            if raycastResult and not raycastResult.Instance:IsDescendantOf(character) then
                continue
            end
        end
        
        if angle < closestAngle then
            closestAngle = angle
            closestTarget = {
                Player = player,
                Part = targetPart,
                Position = targetPart.Position
            }
        end
    end
    
    return closestTarget
end

-- 5. تحديث النظام
RunService.RenderStepped:Connect(function()
    -- تحديث الدائرة
    FOVCircle.Radius = Settings.FOV * 3
    FOVCircle.Position = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    FOVCircle.Visible = Settings.Enabled
    
    -- البحث عن الهدف
    CurrentTarget = FindTarget()
end)

-- 6. وظيفة التصويب المعدلة
function GetModifiedDirection(origin, direction)
    if not Settings.Enabled or not CurrentTarget then
        return direction
    end
    
    local targetPos = CurrentTarget.Part.Position
    
    -- تطبيق التنبؤ
    if Settings.Prediction and CurrentTarget.Part:IsA("BasePart") then
        local velocity = CurrentTarget.Part.Velocity
        local distance = (targetPos - origin).Magnitude
        targetPos = targetPos + (velocity * (distance / 2000))
    end
    
    -- تطبيق التمهيد
    local newDirection = (targetPos - origin).Unit
    return direction:Lerp(newDirection, Settings.Smoothing)
end
